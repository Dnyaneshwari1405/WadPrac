src/app/user.spec.ts

import { Injectable } from '@angular/core';  //“Injectable is used to create a service that can be used across components.”

@Injectable({           //“providedIn: root makes the service available throughout the application.”
  providedIn: 'root'     //@Injectable Decorator
})
export class UserService {  👉 Defines the service class

  userData: any;   //“This variable stores user data temporarily.” 

  register(user: any) {   //“This method stores user data received from the register component.”
    this.userData = user;
  }

  getUser() {
    return this.userData;   /“This method returns stored user data to other components.”
  }
}


src/app/register/register.ts

import { Component } from '@angular/core'; //“Component is used to create UI part of Angular application.”
import { FormsModule } from '@angular/forms'; //“FormsModule is used for two-way data binding.”
import { UserService } from '../user.service'; //👉 To store user data
import { Router } from '@angular/router'; //👉 Used for navigation (page change)

@Component({                        👉 Defines component settings:  
  selector: 'app-register',           selector → HTML tag name ,
  standalone: true,                  standalone: true → no module needed , 
  imports: [FormsModule],         //imports → use FormsModule
  templateUrl: './register.html'    templateUrl → HTML file
})
export class RegisterComponent {         //👉 Main logic of component

  user = { name: '', email: '', password: '' };     👉 Stores input values

  constructor(private userService: UserService, private router: Router) {}   //👉 Injects:
                                                                  //UserService → for storing data ,  //Router → for navigation
  register() {
    this.userService.register(this.user);   
    alert("Registration Successful");
    this.router.navigate(['/login']);     
  }
}
